// ========== Estimate Builder ==========
function EstimateBuilder() {
  const [yards, setYards] = useState(7);
  const truckCapacity = 14; // industry average truck size
  const low = yards * 45;
  const high = yards * 60;
  const loadFraction = yards / truckCapacity;

  const washers = Array.from({ length: yards }, (_, i) => i);

  const presets = [
    { label: "¬º Truck", value: Math.round(truckCapacity * 0.25) },
    { label: "¬Ω Truck", value: Math.round(truckCapacity * 0.5) },
    { label: "¬æ Truck", value: Math.round(truckCapacity * 0.75) },
    { label: "Full Truck", value: truckCapacity },
  ];

  return (
    <div className="bg-white shadow rounded-lg p-6">
      <h3 className="font-bold text-lg mb-3">Estimate Builder</h3>
      <p className="text-sm mb-4 text-gray-700">
        Junk removal is usually priced by cubic yard (about the size of a washing
        machine). Most trucks hold 12‚Äì15 cubic yards ‚Äî about 14 washing machines.
      </p>

      {/* Preset Buttons */}
      <div className="flex gap-2 mb-4 flex-wrap">
        {presets.map((p) => (
          <button
            key={p.label}
            onClick={() => setYards(p.value)}
            className={`px-3 py-1 rounded-md text-sm font-medium ${
              yards === p.value
                ? "bg-green-700 text-white"
                : "bg-gray-200 text-gray-700 hover:bg-gray-300"
            }`}
          >
            {p.label}
          </button>
        ))}
      </div>

      {/* Truck Graphic */}
      <div className="relative w-full h-40 mb-4 flex items-end">
        {/* Truck cab */}
        <div className="w-20 h-20 bg-gray-700 rounded-md relative">
          <div className="absolute bottom-0 left-2 w-8 h-8 bg-gray-900 rounded-full" />
        </div>

        {/* Truck bed */}
        <div className="flex-1 h-28 border-4 border-gray-400 bg-gray-100 rounded-md flex flex-wrap p-2 relative">
          {washers.map((_, i) => (
            <div
              key={i}
              className="w-1/6 flex justify-center items-center"
            >
              <WasherIcon />
            </div>
          ))}
          {/* Wheels */}
          <div className="absolute -bottom-6 left-1/4 w-10 h-10 bg-gray-900 rounded-full" />
          <div className="absolute -bottom-6 right-1/4 w-10 h-10 bg-gray-900 rounded-full" />
        </div>
      </div>

      {/* Slider */}
      <label className="block text-sm font-semibold mb-1">
        Select load size ({yards} cubic yards)
      </label>
      <input
        type="range"
        min="1"
        max={truckCapacity}
        value={yards}
        onChange={(e) => setYards(parseInt(e.target.value))}
        className="w-full mb-3"
      />

      {/* Pricing */}
      <p className="text-sm">
        Estimated Cost:{" "}
        <span className="font-bold">
          ${low.toLocaleString()} ‚Äì ${high.toLocaleString()}
        </span>
      </p>

      {/* Load fraction text */}
      <p className="text-xs text-gray-600">
        That‚Äôs about{" "}
        {loadFraction <= 0.25
          ? "¬º load"
          : loadFraction <= 0.5
          ? "¬Ω load"
          : loadFraction < 1
          ? "¬æ load"
          : "a full truck"}
        .
      </p>

      {/* CTA */}
      <button className="mt-4 bg-green-700 text-white px-4 py-2 rounded-md hover:bg-green-800">
        Request Real Quote
      </button>
    </div>
  );
}
